def is_ip(addr): 
        ipaddr=addr.split('.') 
        if len(ipaddr)!= 4: 
            return False
        for i in range(4):
            try:
                if int(ipaddr[i])>255 or (ipaddr[i])<0:
                    return False
            except ValueError:
                return False
        return True
print is_ip("127.0.0.1")
